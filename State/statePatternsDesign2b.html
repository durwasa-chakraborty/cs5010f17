      class Ball2class implements Ball2 {
          int x;    // x coordinate of this ball
          int y;    // y coordinate of this ball
          int r;    // radius of this ball, in pixels
      
          // (Java constructor and interpretations omitted)
      
          public boolean intersects (Ball2 b2) {
              // ask other ball for its information
              return coordinatesIntersection (b2.getX(),
                                              b2.getY(),
                                              b2.getR());
          }
      
          public boolean coordinatesIntersect (int x2, int y2, int r2) {
              // do the computation here
              return (sqr (x - x2) + sqr (y - y2)) <= (sqr (r + r2));
          }
      
          private int sqr (int x) { return x * x; }
      
          // be prepared to answer if someone else asks you
          // the same questions about your state
      
          public int getX () { return x; }
          public int getY () { return y; }
          public int getR () { return r; }
      }
