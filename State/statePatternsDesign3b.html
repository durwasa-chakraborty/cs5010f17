      class Ball3class implements Ball3 {
          int x;    // x coordinate of this ball
          int y;    // y coordinate of this ball
          int r;    // radius of this ball, in pixels
      
          // (Java constructor and interpretations omitted)
      
          public boolean intersects (Ball3 b2) {
              // send your data to the other ball
              // and ask it to complete the computation
              return b2.intersectResponder (x, y, r);
          }
      
          public boolean intersectResponder (int x2, int y2, int r2) {
              // if someone asks you to complete the computation,
              // do so
              return (sqr (x - x2) + sqr (y - y2)) <= (sqr (r + r2));
          }
      
          private int sqr (int x) { return x * x; }
      }
